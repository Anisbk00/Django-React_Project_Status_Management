# ðŸš€ Django + MySQL Setup in VS Code

# 1. Create Project Folder and Open in VS Code
mkdir my_django_project
cd my_django_project
code .

# 2. Create and Activate Virtual Environment
python -m venv venv
# On Windows:
venv\Scripts\activate


# 3. Install Django and MySQL Client
pip install django mysqlclient

# If mysqlclient fails, use this instead:
# pip install mysql-connector-python

# 4. Start Django Project
django-admin startproject config .

# 5. Create MySQL Database (Run in MySQL CLI or Workbench)
# Replace 'mydb', 'myuser', 'mypassword' with your own values

CREATE DATABASE mydb CHARACTER SET UTF8;
CREATE USER 'myuser'@'localhost' IDENTIFIED BY 'mypassword';
GRANT ALL PRIVILEGES ON mydb.* TO 'myuser'@'localhost';
FLUSH PRIVILEGES;

# 6. Configure MySQL in config/settings.py

# For mysqlclient:
DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.mysql',
        'NAME': 'mydb',
        'USER': 'myuser',
        'PASSWORD': 'mypassword',
        'HOST': 'localhost',
        'PORT': '3306',
    }
}

# Or, if using mysql-connector-python:
DATABASES = {
    'default': {
        'ENGINE': 'mysql.connector.django',
        'NAME': 'mydb',
        'USER': 'myuser',
        'PASSWORD': 'mypassword',
        'HOST': 'localhost',
        'PORT': '3306',
        'OPTIONS': {
            'autocommit': True,
        }
    }
}

# 7. Apply Migrations
python manage.py migrate

# 8. Create Superuser (for admin panel)
python manage.py createsuperuser

# 9. Run the Development Server
python manage.py runserver
